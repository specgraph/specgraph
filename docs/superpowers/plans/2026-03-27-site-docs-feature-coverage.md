Add to `cmd/specgraph/docs.go`:

````go
func runDocsCli(_ *cobra.Command, args []string) error {
	outPath := docsCliDefaultPath
	if len(args) > 0 {
		outPath = args[0]
	}

	var sb strings.Builder
	writeHeader(&sb)
	writeTOC(&sb)
	writeGroups(&sb, rootCmd)

	if err := os.MkdirAll(filepath.Dir(outPath), 0o755); err != nil {
		return fmt.Errorf("create output dir: %w", err)
	}
	if err := os.WriteFile(outPath, []byte(sb.String()), 0o644); err != nil {
		return fmt.Errorf("write cli reference: %w", err)
	}
	fmt.Fprintf(os.Stderr, "CLI reference written to %s\n", outPath)
	return nil
}

func writeHeader(sb *strings.Builder) {
	sb.WriteString("# CLI Reference\n\n")
	sb.WriteString("> Auto-generated by `specgraph docs cli`. Do not edit manually.\n\n")
	sb.WriteString("---\n\n")
}

func writeTOC(sb *strings.Builder) {
	for _, g := range commandGroups {
		anchor := strings.ToLower(g.Name)
		anchor = strings.NewReplacer(" & ", "-", " ", "-").Replace(anchor)
		fmt.Fprintf(sb, "- [%s](#%s)\n", g.Name, anchor)
	}
	sb.WriteString("\n---\n\n")
}

func writeGroups(sb *strings.Builder, root *cobra.Command) {
	// Build a lookup from command name to *cobra.Command.
	cmdMap := make(map[string]*cobra.Command)
	for _, c := range root.Commands() {
		cmdMap[c.Name()] = c
	}

	written := make(map[string]bool)
	for _, g := range commandGroups {
		fmt.Fprintf(sb, "## %s\n\n", g.Name)
		for _, name := range g.Commands {
			cmd, ok := cmdMap[name]
			if !ok {
				continue
			}
			writeCommand(sb, cmd, "###")
			written[name] = true
		}
	}

	// Collect ungrouped commands into "Other".
	var other []*cobra.Command
	for _, c := range root.Commands() {
		if !written[c.Name()] && c.Name() != "help" && c.Name() != "completion" && c.Name() != "docs" {
			other = append(other, c)
		}
	}
	if len(other) > 0 {
		sb.WriteString("## Other\n\n")
		for _, c := range other {
			writeCommand(sb, c, "###")
		}
	}
}

func writeCommand(sb *strings.Builder, cmd *cobra.Command, heading string) {
	fmt.Fprintf(sb, "%s %s\n\n", heading, cmd.CommandPath())

	if cmd.Long != "" {
		sb.WriteString(cmd.Long + "\n\n")
	} else if cmd.Short != "" {
		sb.WriteString(cmd.Short + "\n\n")
	}

	fmt.Fprintf(sb, "```\n%s\n```\n\n", cmd.UseLine())

	if cmd.HasAvailableLocalFlags() {
		sb.WriteString("**Flags:**\n\n```\n")
		sb.WriteString(cmd.LocalFlags().FlagUsages())
		sb.WriteString("```\n\n")
	}

	if cmd.Example != "" {
		sb.WriteString("**Examples:**\n\n```bash\n")
		sb.WriteString(cmd.Example)
		sb.WriteString("\n```\n\n")
	}

	// Recurse into subcommands.
	for _, sub := range cmd.Commands() {
		if sub.Name() == "help" {
			continue
		}
		writeCommand(sb, sub, heading+"#")
	}
}
````
